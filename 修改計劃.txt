2.1 為什麼選擇 Callable Functions？
-----------------------------------
相比傳統 HTTP Functions，Callable Functions 提供：
✅ 自動處理 Firebase Auth 認證 (不需手動解析 Bearer Token)
✅ 自動處理 CORS (不需 cors 中介軟體)
✅ 客戶端調用更簡單 (使用 Firebase SDK)
✅ 自動序列化/反序列化 JSON
✅ 內建錯誤處理機制
✅ 支援離線持久化

3.2.1 getTask - 獲取單一任務 ⭐️ 新增

export const getTask = onCall(
  { region: "asia-east1" },
  async (request: CallableRequest) => {
    if (!request.auth) {
      throw new Error("需要登入");
    }

    const userId = request.auth.uid;
    const { taskId } = request.data;

    if (!taskId) {
      throw new Error("缺少任務 ID");
    }

    try {
      const taskDoc = await db.collection("tasks")
        .doc(userId)
        .collection("userTasks")
        .doc(taskId)
        .get();

      if (!taskDoc.exists) {
        throw new Error("任務不存在");
      }

      // 獲取任務實例
      const instancesSnapshot = await taskDoc.ref
        .collection("instances")
        .get();

      const instances = instancesSnapshot.docs.map((doc) => ({
        id: doc.id,
        ...doc.data(),
      }));

      return {
        success: true,
        task: {
          id: taskDoc.id,
          ...taskDoc.data(),
          instances,
        },
      };
    } catch (error) {
      console.error("獲取任務失敗:", error);
      throw new Error("獲取任務失敗");
    }
  }
);

export const toggleTaskCompletion = onCall(
  { region: "asia-east1" },
  async (request: CallableRequest) => {
    if (!request.auth) {
      throw new Error("需要登入");
    }

    const userId = request.auth.uid;
    const { taskId, instanceId, isCompleted } = request.data;

    if (!taskId) {
      throw new Error("缺少任務 ID");
    }

    try {
      if (instanceId) {
        // 更新實例
        const instanceRef = db.collection("tasks")
          .doc(userId)
          .collection("userTasks")
          .doc(taskId)
          .collection("instances")
          .doc(instanceId);

        await instanceRef.update({
          isCompleted,
          updatedAt: admin.firestore.FieldValue.serverTimestamp(),
        });
      } else {
        // 更新主任務
        const taskRef = db.collection("tasks")
          .doc(userId)
          .collection("userTasks")
          .doc(taskId);

        await taskRef.update({
          isCompleted,
          updatedAt: admin.firestore.FieldValue.serverTimestamp(),
        });
      }

      return { success: true };
    } catch (error) {
      console.error("切換完成狀態失敗:", error);
      throw new Error("切換完成狀態失敗");
    }
  }
);

export const updateCategoryStats = onCall(
  { region: "asia-east1" },
  async (request: CallableRequest) => {
    if (!request.auth) {
      throw new Error("需要登入");
    }

    const userId = request.auth.uid;
    const { category, updates } = request.data;

    if (!category || !updates) {
      throw new Error("缺少必要參數");
    }

    try {
      // 查找該類別的統計資料
      const statsSnapshot = await db.collection("userStatistics")
        .doc(userId)
        .collection("statistics")
        .where("category", "==", category)
        .limit(1)
        .get();

      let statRef;

      if (statsSnapshot.empty) {
        // 建立新的統計資料
        statRef = db.collection("userStatistics")
          .doc(userId)
          .collection("statistics")
          .doc();

        await statRef.set({
          userId,
          category,
          progress: updates.progress || 0,
          taskcount: updates.taskCount || 0,
          taskcompletecount: updates.taskCompleteCount || 0,
          totalFocusTime: updates.focusTime || 0,
          date: admin.firestore.FieldValue.serverTimestamp(),
          updatedAt: admin.firestore.FieldValue.serverTimestamp(),
        });
      } else {
        // 更新現有統計資料
        statRef = statsSnapshot.docs[0].ref;

        const updateData: any = {
          updatedAt: admin.firestore.FieldValue.serverTimestamp(),
        };

        if (updates.progress !== undefined) {
          updateData.progress = updates.progress;
        }

        if (updates.focusTime !== undefined) {
          updateData.totalFocusTime = admin.firestore.FieldValue.increment(updates.focusTime);
        }

        if (updates.taskCount !== undefined) {
          updateData.taskcount = admin.firestore.FieldValue.increment(updates.taskCount);
        }

        if (updates.taskCompleteCount !== undefined) {
          updateData.taskcompletecount = admin.firestore.FieldValue.increment(updates.taskCompleteCount);
        }

        await statRef.update(updateData);
      }

      return { success: true };
    } catch (error) {
      console.error("更新類別統計失敗:", error);
      throw new Error("更新類別統計失敗");

// 驗證輔助函數
export function validateData<T>(
  schema: z.ZodSchema<T>,
  data: unknown
): { success: true; data: T } | { success: false; errors: string[] } {
  try {
    const validatedData = schema.parse(data);
    return { success: true, data: validatedData };
  } catch (error) {
    if (error instanceof z.ZodError) {
      return {
        success: false,
        errors: error.errors.map((e) => e.message),
      };
    }
    return {
      success: false,
      errors: ["驗證失敗"],
    };
  }
}

6.2.2 建立自訂錯誤類別 (src/utils/errors.ts)

import { HttpsError } from "firebase-functions/v2/https";

// 自訂錯誤類別
export class APIError extends HttpsError {
  constructor(
    code: string,
    message: string,
    public details?: any
  ) {
    super(code as any, message, details);
    this.name = "APIError";
  }
}

// 錯誤工廠函數
export const Errors = {
  // 認證錯誤
  Unauthenticated: (message = "需要登入") =>
    new APIError("unauthenticated", message),

  // 權限錯誤
  PermissionDenied: (message = "權限不足") =>
    new APIError("permission-denied", message),

  // 驗證錯誤
  InvalidArgument: (message: string, details?: any) =>
    new APIError("invalid-argument", message, details),

  // 資源不存在
  NotFound: (resource: string) =>
    new APIError("not-found", `${resource}不存在`),

  // 資源已存在
  AlreadyExists: (resource: string) =>
    new APIError("already-exists", `${resource}已存在`),

  // 內部錯誤
  Internal: (message = "內部伺服器錯誤", error?: Error) => {
    console.error("Internal error:", error);
    return new APIError("internal", message);
  },

  // 服務不可用
  Unavailable: (message = "服務暫時不可用") =>
    new APIError("unavailable", message),
};

import { Errors } from "./utils/errors";
import { validateData, TaskSchema } from "./utils/validation";

export const updateTask = onCall(
  { region: "asia-east1" },
  async (request: CallableRequest) => {
    // 認證檢查
    if (!request.auth) {
      throw Errors.Unauthenticated();
    }

    const userId = request.auth.uid;
    const { taskId, updates } = request.data;

    // 參數檢查
    if (!taskId) {
      throw Errors.InvalidArgument("缺少任務 ID");
    }

    if (!updates) {
      throw Errors.InvalidArgument("缺少更新資料");
    }

    // 資料驗證
    const validation = validateData(TaskSchema.partial(), updates);
    if (!validation.success) {
      throw Errors.InvalidArgument(
        "資料驗證失敗",
        { errors: validation.errors }
      );
    }

    try {
      const taskRef = db.collection("tasks")
        .doc(userId)
        .collection("userTasks")
        .doc(taskId);

      // 檢查任務是否存在
      const taskDoc = await taskRef.get();
      if (!taskDoc.exists) {
        throw Errors.NotFound("任務");
      }

      // 更新任務
      await taskRef.update({
        ...validation.data,
        updatedAt: admin.firestore.FieldValue.serverTimestamp(),
      });

      return { success: true };
    } catch (error) {
      // 如果是已知錯誤,直接拋出
      if (error instanceof APIError) {
        throw error;
      }

      // 未知錯誤,包裝成 Internal Error
      throw Errors.Internal("更新任務失敗", error as Error);
    }
  }
);

import * as logger from "firebase-functions/logger";

export const Logger = {
  // 一般資訊
  info: (message: string, data?: any) => {
    logger.info(message, data);
  },

  // 警告
  warn: (message: string, data?: any) => {
    logger.warn(message, data);
  },

  // 錯誤
  error: (message: string, error: Error, data?: any) => {
    logger.error(message, {
      error: {
        name: error.name,
        message: error.message,
        stack: error.stack,
      },
      ...data,
    });
  },

  // 函數執行追蹤
  trace: (functionName: string, userId: string, action: string, data?: any) => {
    logger.info(`[${functionName}] ${action}`, {
      userId,
      timestamp: new Date().toISOString(),
      ...data,
    });
  },
};

6.3 錯誤處理最佳實踐
--------------------

✅ 總是驗證輸入資料
✅ 使用標準化的錯誤代碼
✅ 提供有意義的錯誤訊息

❌ 不要在錯誤訊息中暴露內部實作細節
❌ 不要忽略錯誤或使用空的 catch 區塊
❌ 不要使用通用的錯誤訊息 (如 "發生錯誤")

import FirebaseCore

class CloudFunctionService {
    private let functions = Functions.functions(region: "asia-east1")

    // 建立任務
    func createTask(_ task: TodoTask) async throws -> String {
        let callable = functions.httpsCallable("createTask")

        let data: [String: Any] = [
            "task": task.toFirestore()
        ]

        let result = try await callable.call(data)

        guard let response = result.data as? [String: Any],
              let taskId = response["taskId"] as? String else {
            throw NSError(domain: "CloudFunction", code: -1)
        }

        return taskId
    }

    // 獲取任務列表
    func fetchTasks() async throws -> [TodoTask] {
        let callable = functions.httpsCallable("fetchTasks")
        let result = try await callable.call()

        guard let response = result.data as? [String: Any],
              let tasksData = response["tasks"] as? [[String: Any]] else {
            throw NSError(domain: "CloudFunction", code: -1)
        }

        return tasksData.compactMap { TodoTask(dictionary: $0) }
    }

    // 更新任務
    func updateTask(taskId: String, updates: [String: Any]) async throws {
        let callable = functions.httpsCallable("updateTask")

        let data: [String: Any] = [
            "taskId": taskId,
            "updates": updates
        ]

        _ = try await callable.call(data)
    }

    // 切換完成狀態
    func toggleTaskCompletion(
        taskId: String,
        instanceId: String?,
        isCompleted: Bool
    ) async throws {
        let callable = functions.httpsCallable("toggleTaskCompletion")

        var data: [String: Any] = [
            "taskId": taskId,
            "isCompleted": isCompleted
        ]

        if let instanceId = instanceId {
            data["instanceId"] = instanceId
        }

        _ = try await callable.call(data)
    }
}

import FirebaseFunctions

func handleCloudFunctionError(_ error: Error) -> String {
    guard let functionsError = error as? FunctionsError else {
        return "未知錯誤"
    }

    switch functionsError.code {
    case .unauthenticated:
        return "請先登入"
    case .permissionDenied:
        return "權限不足"
    case .invalidArgument:
        return "無效的參數"
    case .notFound:
        return "資源不存在"
    case .alreadyExists:
        return "資源已存在"
    case .internal:
        return "伺服器錯誤,請稍後再試"
    case .unavailable:
        return "服務暫時不可用"
    default:
        return "發生錯誤: \(functionsError.localizedDescription)"
    }
}

修改 1: 添加實例生成輔助函數
--------------------------------------------------------------------------------
位置: functions/src/index.ts (在現有函數之前)
目的: 計算重複任務的所有實例日期

/**
 * 計算重複任務的實例日期
 * @param task 任務資料
 * @param limit 最大實例數量（預設 100）
 * @returns Date 陣列
 */
function calculateNextOccurrences(task: any, limit: number = 100): Date[] {
  const occurrences: Date[] = [];

  // 確保不超過結束日期
  const endDate = task.repeatEndDate || new Date(
    task.startDate.getTime() + 365 * 24 * 60 * 60 * 1000 // 一年後
  );

  // 先加入開始日期（當天）
  occurrences.push(new Date(task.startDate));

  let currentDate = new Date(task.startDate);

  while (occurrences.length < limit) {
    if (currentDate > endDate) {
      break;
    }

    let nextDate: Date | null = null;

    switch (task.repeatType?.type) {
      case 'daily':
        nextDate = new Date(currentDate);
        nextDate.setDate(nextDate.getDate() + 1);
        break;

      case 'weekly':
        // 使用創建時的星期幾
        nextDate = new Date(currentDate);
        nextDate.setDate(nextDate.getDate() + 7);
        break;

      case 'monthly':
        // 使用創建時的日期
        const dayOfMonth = task.startDate.getDate();
        nextDate = new Date(currentDate);
        nextDate.setMonth(nextDate.getMonth() + 1);
        nextDate.setDate(dayOfMonth);
        break;

      case 'none':
      default:
        return occurrences;
    }

    if (nextDate && nextDate <= endDate) {
      occurrences.push(nextDate);
      currentDate = nextDate;
    } else {
      break;
    }
  }

  return occurrences;
}

修改 2: 添加檢查實例重新生成的函數

/**
 * 檢查是否需要重新生成實例
 * @param taskRef 任務文檔引用
 * @param newTask 新任務資料
 * @returns boolean
 */
async function needsRegenerateInstances(
  taskRef: FirebaseFirestore.DocumentReference,
  newTask: any
): Promise<boolean> {
  try {
    const doc = await taskRef.get();

    if (!doc.exists) {
      return true; // 新任務，需要生成實例
    }

    const oldData = doc.data();
    if (!oldData) {
      return true;
    }

    // 檢查重複類型是否改變
    const oldRepeatType = oldData.repeatType?.type || 'none';
    const newRepeatType = newTask.repeatType?.type || 'none';
    if (oldRepeatType !== newRepeatType) {
      return true;
    }

    // 檢查開始時間是否改變
    if (oldData.startDate?.toDate?.()?.getTime() !== newTask.startDate?.getTime?.()) {
      return true;
    }

    // 檢查結束時間是否改變
    const oldEndTime = oldData.endDate?.toDate?.()?.getTime?.();
    const newEndTime = newTask.endDate?.getTime?.();
    if (oldEndTime !== newEndTime) {
      return true;
    }

    // 檢查重複結束時間是否改變
    const oldRepeatEndTime = oldData.repeatEndDate?.toDate?.()?.getTime?.();
    const newRepeatEndTime = newTask.repeatEndDate?.getTime?.();
    if (oldRepeatEndTime !== newRepeatEndTime) {
      return true;
    }

    return false;
  } catch (error) {
    console.error('Error checking task changes:', error);
    return true; // 如果出錯，為安全起見重新生成實例
  }
}

```typescript
export const createTask = onCall(
  { region: "asia-east1" },
  async (request: CallableRequest) => {
    if (!request.auth) {
      throw new Error("需要登入");
    }

    const userId = request.auth.uid;
    const taskData = request.data.task;

    if (!taskData || !taskData.title) {
      throw new Error("無效的任務資料");
    }

    try {
      const batch = db.batch();

      // 建立任務文檔引用
      const taskRef = db.collection("tasks")
        .doc(userId)
        .collection("userTasks")
        .doc(); // 使用 doc() 生成 ID

      // 準備任務資料
      const taskDataToSave = {
        ...taskData,
        userId,
        createdAt: admin.firestore.FieldValue.serverTimestamp(),
        updatedAt: admin.firestore.FieldValue.serverTimestamp(),
      };

      // 處理重複任務的結束日期
      if (taskData.repeatType?.type !== 'none') {
        if (taskData.repeatEndDate) {
          taskDataToSave.repeatEndDate = taskData.repeatEndDate;
        } else {
          // 如果沒有設定結束日期，預設為一年後
          const oneYearLater = new Date(taskData.startDate);
          oneYearLater.setFullYear(oneYearLater.getFullYear() + 1);
          taskDataToSave.repeatEndDate = admin.firestore.Timestamp.fromDate(oneYearLater);
        }
      }

      // 設置任務資料
      batch.set(taskRef, taskDataToSave);

      // 如果是重複性任務，生成實例
      if (taskData.repeatType?.type !== 'none') {
        // 準備用於計算的任務物件
        const taskForCalculation = {
          startDate: taskData.startDate instanceof admin.firestore.Timestamp
            ? taskData.startDate.toDate()
            : new Date(taskData.startDate),
          endDate: taskData.endDate instanceof admin.firestore.Timestamp
            ? taskData.endDate.toDate()
            : new Date(taskData.endDate),
          repeatType: taskData.repeatType,
          repeatEndDate: taskDataToSave.repeatEndDate instanceof admin.firestore.Timestamp
            ? taskDataToSave.repeatEndDate.toDate()
            : taskDataToSave.repeatEndDate
              ? new Date(taskDataToSave.repeatEndDate)
              : null,
        };

        const occurrences = calculateNextOccurrences(taskForCalculation, 100);

        for (const date of occurrences) {
          const instanceRef = taskRef.collection("instances").doc();
          batch.set(instanceRef, {
            date: admin.firestore.Timestamp.fromDate(date),
            isCompleted: false,
            parentTaskId: taskRef.id,
            createdAt: admin.firestore.FieldValue.serverTimestamp(),
          });
        }
      }

      await batch.commit();

      return {
        success: true,
        taskId: taskRef.id,
      };
    } catch (error) {
      console.error("創建任務失敗:", error);
      throw new Error("創建任務失敗");
```

```typescript
export const updateTask = onCall(
  { region: "asia-east1" },
  async (request: CallableRequest) => {
    if (!request.auth) {
      throw new Error("需要登入");
    }

    const userId = request.auth.uid;
    const { taskId, updates } = request.data;

    if (!taskId || !updates) {
      throw new Error("缺少必要參數");
    }

    try {
      const batch = db.batch();
      const taskRef = db.collection("tasks")
        .doc(userId)
        .collection("userTasks")
        .doc(taskId);

      // 準備更新的任務資料（用於檢查）
      const taskForCheck = {
        startDate: updates.startDate instanceof admin.firestore.Timestamp
          ? updates.startDate.toDate()
          : updates.startDate ? new Date(updates.startDate) : null,
        endDate: updates.endDate instanceof admin.firestore.Timestamp
          ? updates.endDate.toDate()
          : updates.endDate ? new Date(updates.endDate) : null,
        repeatType: updates.repeatType,
        repeatEndDate: updates.repeatEndDate instanceof admin.firestore.Timestamp
          ? updates.repeatEndDate.toDate()
          : updates.repeatEndDate ? new Date(updates.repeatEndDate) : null,
      };

      // 檢查是否需要重新生成實例
      const needsRegenerate = await needsRegenerateInstances(taskRef, taskForCheck);

      if (needsRegenerate && updates.repeatType?.type !== 'none') {
        // 獲取所有現有的實例
        const instancesSnapshot = await taskRef.collection("instances").get();
        const completedInstanceDates: Date[] = [];

        // 保存已完成實例的日期，刪除未完成的實例
        for (const instanceDoc of instancesSnapshot.docs) {
          const data = instanceDoc.data();
          const isCompleted = data.isCompleted || false;

          if (isCompleted) {
            if (data.date) {
              completedInstanceDates.push(data.date.toDate());
            }
          } else {
            // 刪除未完成的實例
            batch.delete(instanceDoc.ref);
          }
        }

        // 生成新的實例
        const occurrences = calculateNextOccurrences(taskForCheck, 100);

        for (const date of occurrences) {
          // 檢查這個日期是否已經有完成的實例
          const isDateCompleted = completedInstanceDates.some(completedDate => {
            return completedDate.toDateString() === date.toDateString();
          });

          // 只為未完成的日期生成新實例
          if (!isDateCompleted) {
            const instanceRef = taskRef.collection("instances").doc();
            batch.set(instanceRef, {
              date: admin.firestore.Timestamp.fromDate(date),
              isCompleted: false,
              parentTaskId: taskId,
              createdAt: admin.firestore.FieldValue.serverTimestamp(),
            });
          }
        }
      }

      // 更新任務本身
      batch.update(taskRef, {
        ...updates,
        updatedAt: admin.firestore.FieldValue.serverTimestamp(),
      });

      await batch.commit();

      return { success: true };
    } catch (error) {
      console.error("更新任務失敗:", error);
      throw new Error("更新任務失敗");
    }
  }
);
```

⚠️ 重要提醒
================================================================================

1. **日期處理**: 注意 Firebase Timestamp 和 JavaScript Date 的轉換
   - 客戶端傳來的可能是 ISO 字符串或 Timestamp
   - 需要統一轉換為 Date 物件進行計算
   - 存入 Firestore 時使用 admin.firestore.Timestamp.fromDate()

2. **Batch 操作限制**: Firestore batch 最多 500 個操作
   - 如果實例超過 490 個，需要分批處理
   - 建議限制 limit 參數不超過 100

3. **錯誤處理**: 確保所有錯誤都有適當的 try-catch
   - 實例生成失敗不應導致任務創建失敗
   - 考慮記錄詳細日誌

4. **向後兼容**: 現有任務可能沒有 repeatType 欄位
   - 需要處理 undefined 情況
   - 預設為 { type: 'none' }
