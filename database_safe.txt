//Firestore 安全規則草案
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // 允許認證過的使用者，讀寫自己的 user profile
    match /users/{userId} {
      allow read, write: if request.auth != null && request.auth.uid == userId;

      // User 子集合: todos
      match /todos/{todoId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }

      // User 子集合: chatRooms
      match /chatRooms/{chatRoomId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;

        // 子子集合: messages
        match /messages/{messageId} {
          allow read, write: if request.auth != null && request.auth.uid == userId;
        }
      }

      // User 子集合: timerRecords
      match /timerRecords/{timerRecordId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }

      // User 子集合: learningStatistics
      match /learningStatistics/{statisticId} {
        allow read, write: if request.auth != null && request.auth.uid == userId;
      }
    }
  }
}
